# Preprocessing and Model Building

from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.pipeline import Pipeline
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from xgboost import XGBClassifier
from sklearn.metrics import classification_report
In[]:
# Features and target
target = 'Quit_the_Company'
ignore_features = ['employee_id']
categorical_features = ['Departments', 'salary']
In[]:
# Define X and y
X = df_hr_data.drop(columns=[target] + ignore_features)
y = df_hr_data[target]
In[]:
# Preprocessing pipeline
numeric_features = X.select_dtypes(include=['int64', 'float64']).columns.tolist()
categorical_features = categorical_features

preprocessor = ColumnTransformer(transformers=[
    ('num', StandardScaler(), numeric_features),
    ('cat', OneHotEncoder(handle_unknown='ignore'), categorical_features)
])
In[]:
# Model pipeline using XGBoost
model = Pipeline(steps=[
    ('preprocessor', preprocessor),
    ('classifier', XGBClassifier(use_label_encoder=False, eval_metric='logloss', random_state=123))
])
In[]:
# Split data
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=123)
In[]:
# Train the model
model.fit(X_train, y_train)

# Plotting feature importance

import matplotlib.pyplot as plt
import seaborn as sns
import pandas as pd
import numpy as np





















In[]:






# Extract the trained XGBoost classifier from the pipeline
classifier = model.named_steps['classifier']
preprocessor = model.named_steps['preprocessor']





















In[]:






# Get the feature names after preprocessing
# numeric features first, then encoded categorical features
num_features = numeric_features
cat_features = list(preprocessor.named_transformers_['cat'].get_feature_names_out(categorical_features))
all_features = num_features + cat_features





















In[]:






# Get feature importances from the model
importances = classifier.feature_importances_





















In[]:






# Create DataFrame with features and their importance
feature_table = pd.DataFrame({
    'feature': all_features,
    'importance': importances
}).sort_values(by='importance', ascending=False)





















In[]:






# Plot the feature importance
plt.figure(figsize=(10, 6))
sns.barplot(x='importance', y='feature', data=feature_table)
plt.title("Feature Importance")
plt.tight_layout()
plt.show()
